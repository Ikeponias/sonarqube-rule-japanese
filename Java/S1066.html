折りたたみ可能なif ステートメントをマージすることで、コードの可読性を高めています。

コンプライアンス違反のコード例

if (file != null) {
  if (file.isFile() || file.isDirectory()) {
    /* ... */
  }
}

コンプライアンス・ソリューション

if (file != null && isFileOrDirectory(file)) {
  /* ... */
}

private static boolean isFileOrDirectory(File file) {
  return file.isFile() || file.isDirectory();
}
